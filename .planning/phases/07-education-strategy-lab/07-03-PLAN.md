```typescript
interface Strategy {
  id: string;
  userId: string;
  name: string;
  entryRules: EntryRule[];
  exitRules: ExitRule[];
  riskRules: RiskRule[];
  createdAt: Date;
  lastTestedAt?: Date;
}

interface EntryRule {
  condition: 'price_above' | 'price_below' | 'ma_cross' | 'rsi_above' | 'rsi_below' | 'custom';
  value: string;
  timeframes: ('M5' | 'M15' | 'H1' | 'H4' | 'D1')[];
}

interface ExitRule {
  type: 'tp' | 'sl' | 'trailing' | 'time';
  value: number;
  unit: 'pips' | 'percent' | 'atr';
}

interface RiskRule {
  type: 'fixed_lot' | 'percent_balance' | 'kelly' | 'atr_based';
  value: number;
  maxDrawdownPercent: number;
}
```

From supabase/migrations (per 07-CONTEXT.md):
```sql
CREATE TABLE strategies (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id UUID REFERENCES auth.users(id) ON DELETE CASCADE,
  name TEXT NOT NULL,
  entry_rules JSONB NOT NULL,
  exit_rules JSONB NOT NULL,
  risk_rules JSONB NOT NULL,
  created_at TIMESTAMPTZ DEFAULT NOW(),
  last_tested_at TIMESTAMPTZ
);
```
