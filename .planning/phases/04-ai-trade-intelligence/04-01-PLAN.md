// Trade quality analysis (INTL-04)
interface TradeQualityAnalysis {
  entryTimingScore: number;     // 0-100
  exitQualityScore: number;     // 0-100
  rrEfficiency: number;         // Actual RR vs available RR
  qualityGrade: 'A' | 'B' | 'C' | 'D';
}

// Behavioral patterns (INTL-05)
interface BehavioralPatterns {
  revengeTrading: { detected: boolean; severity: 'low' | 'medium' | 'high'; evidence: string[]; tradeIds: string[] };
  impulsiveTrades: { detected: boolean; count: number; triggers: string[] };
  overconfidence: { detected: boolean; evidence: string[] };
  fearExits: { detected: boolean; instances: string[] };
  overtrading: { detected: boolean; sessionCount: number };
  fomo: { detected: boolean; evidence: string[] };
}

// Risk metrics (INTL-06)
interface RiskMetrics {
  lotSizeVariance: number;
  drawdownBehavior: DrawdownProfile;
  exposureProfile: ExposureSnapshot;
  marginPressureEvents: number;
}

interface DrawdownProfile {
  maxDrawdown: number;
  maxDrawdownDuration: number;
  avgRecoveryDays: number;
  recoveryPattern: 'slow' | 'normal' | 'fast';
}

// Performance analytics (INTL-07)
interface PerformanceAnalytics {
  bestSessions: SessionSummary[];
  worstConditions: ConditionSummary[];
  pairPerformance: Record<string, PairStats>;
  timeAnalysis: TimeHeatmap;
}

interface PairStats {
  symbol: string;
  winRate: number;
  avgRR: number;
  totalTrades: number;
  totalPnl: number;
  bestSession: string;
}

// AI insights (INTL-08)
interface AIInsight {
  id: string;
  type: 'behavioral' | 'risk' | 'pattern' | 'action';
  title: string;
  description: string;
  confidence: number;
  actionability: 'quick_fix' | 'requires_attention' | 'strategic';
  suggestedAction: string;
  relatedTrades: string[];
}

Implement rule-based behavioral pattern detection as per CONTEXT.md:

1. **revenge.ts**: Detect revenge trading (3+ trades after 2 consecutive losses within same session, with position size increase >=1.5x within 10 min)

2. **impulsive.ts**: Detect impulsive trades (<5 min hold during high volatility, entry during news events)

3. **overconfidence.ts**: Detect overconfidence (5+ consecutive wins followed by lot size increase >30%)

4. **fomo.ts**: Detect FOMO trades (high volatility entry without pre-trade plan adherence <3)

5. **overtrading.ts**: Detect overtrading (>8 trades in single session)

6. **detector.ts**: Orchestrate all pattern detectors, aggregate results into BehavioralPatterns interface

Each detector should:
- Accept Trade[] and return pattern-specific result
- Include evidence strings explaining detection rationale
- Mark severity levels (low/medium/high)
- Only flag patterns with minimum 3 trades (per research anti-patterns)
