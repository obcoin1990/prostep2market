interface PairStat {
  symbol: string;
  winRate: number;
  avgRR: number;
  totalTrades: number;
  totalPnl: number;
  bestSession: string;
}

interface TimeStat {
  session: 'asian' | 'london' | 'newyork' | 'sydney';
  winRate: number;
  avgRR: number;
  totalTrades: number;
}

export function TradeStatsWidget({ pairStats, timeStats }: { 
  pairStats: PairStat[]; 
  timeStats: TimeStat[];
}) {
  // Sort pairs by win rate descending, take top 5
  const topPairs = [...pairStats]
    .sort((a, b) => b.winRate - a.winRate)
    .slice(0, 5);

  // Sort sessions by win rate
  const bestTimes = [...timeStats].sort((a, b) => b.winRate - a.winRate);
  const worstTimes = [...timeStats].sort((a, b) => a.winRate - b.winRate);

  return (
    <Card>
      <h3 className="text-lg font-semibold mb-4">Trade Statistics</h3>
      
      {/* Top pairs table */}
      <div className="mb-6">
        <h4 className="text-sm font-medium mb-2">Top Performing Pairs</h4>
        <Table>
          <TableHeader>
            <TableRow>
              <TableHead>Pair</TableHead>
              <TableHead>Win Rate</TableHead>
              <TableHead>Avg RR</TableHead>
              <TableHead>Trades</TableHead>
            </TableRow>
          </TableHeader>
          <TableBody>
            {topPairs.map(pair => (
              <TableRow key={pair.symbol}>
                <TableCell className="font-medium">{pair.symbol}</TableCell>
                <TableCell>
                  <span className={pair.winRate >= 50 ? 'text-green' : 'text-red'}>
                    {pair.winRate.toFixed(1)}%
                  </span>
                </TableCell>
                <TableCell>{pair.avgRR.toFixed(2)}</TableCell>
                <TableCell>{pair.totalTrades}</TableCell>
              </TableRow>
            ))}
          </TableBody>
        </Table>
      </div>

      {/* Best/worst times */}
      <div className="grid grid-cols-2 gap-4">
        <div>
          <h4 className="text-sm font-medium mb-2">Best Times</h4>
          {bestTimes.slice(0, 2).map(t => (
            <div key={t.session} className="flex justify-between text-sm">
              <span className="capitalize">{t.session}</span>
              <span className="text-green">{t.winRate.toFixed(0)}%</span>
            </div>
          ))}
        </div>
        <div>
          <h4 className="text-sm font-medium mb-2">Worst Times</h4>
          {worstTimes.slice(0, 2).map(t => (
            <div key={t.session} className="flex justify-between text-sm">
              <span className="capitalize">{t.session}</span>
              <span className="text-red">{t.winRate.toFixed(0)}%</span>
            </div>
          ))}
        </div>
      </div>
    </Card>
  );
}

```tsx
interface HeatmapCell {
  dayOfWeek: number; // 0 = Sunday
  session: 'asian' | 'london' | 'newyork' | 'sydney';
  totalTrades: number;
  totalPnl: number;
  winRate: number;
  avgRR: number;
}

const SESSIONS = ['asian', 'london', 'newyork', 'sydney'] as const;
const DAYS = ['Sun', 'Mon', 'Tue', 'Wed', 'Thu', 'Fri', 'Sat'];

export function SessionHeatmap({ data }: { data: HeatmapCell[] }) {
  // Color gradient: red (loss) -> white (breakeven) -> green (profit)
  const getCellColor = (pnl: number) => {
    const maxPnL = Math.max(...data.map(c => Math.abs(c.totalPnl)), 1000);
    const intensity = Math.min(Math.abs(pnl) / maxPnL, 1);
    
    if (pnl < 0) {
      return `rgba(229, 57, 53, ${intensity})`; // Red with opacity
    } else if (pnl > 0) {
      return `rgba(46, 125, 50, ${intensity})`; // Green with opacity
    }
    return 'rgba(255, 255, 255, 0)';
  };

  // Build 7x4 grid
  const grid = DAYS.map((day, dayIndex) => (
    SESSIONS.map(session => {
      const cell = data.find(c => c.dayOfWeek === dayIndex && c.session === session);
      return {
        day,
        session,
        ...cell
      };
    })
  ));

  return (
    <Card>
      <h3 className="text-lg font-semibold mb-4">Session Analytics</h3>
      
      <div className="overflow-x-auto">
        <table className="w-full">
          <thead>
            <tr>
              <th className="text-left text-sm font-medium p-2"></th>
              {SESSIONS.map(s => (
                <th key={s} className="text-center text-sm font-medium p-2 capitalize">
                  {s}
                </th>
              ))}
            </tr>
          </thead>
          <tbody>
            {grid.map((row, i) => (
              <tr key={i}>
                <td className="text-sm font-medium p-2">{row[0].day}</td>
                {row.map((cell, j) => (
                  <td key={j} className="p-1">
                    <div
                      className="rounded p-2 text-center cursor-pointer hover:opacity-80 transition-opacity"
                      style={{ backgroundColor: cell.totalPnl !== undefined 
                        ? getCellColor(cell.totalPnl) 
                        : '#f5f5f5' 
                      }}
                      title={cell.totalPnl !== undefined 
                        ? `${cell.session}: ${cell.totalTrades} trades, P&L: $${cell.totalPnl.toFixed(0)}, WR: ${cell.winRate.toFixed(0)}%`
                        : 'No data'
                      }
                    >
                      {cell.totalTrades > 0 ? (
                        <div className="text-xs">
                          <div className="font-medium">{cell.totalTrades}</div>
                          <div className={cell.totalPnl >= 0 ? 'text-green' : 'text-red'}>
                            ${cell.totalPnl.toFixed(0)}
                          </div>
                        </div>
                      ) : (
                        <span className="text-gray-400 text-xs">-</span>
                      )}
                    </div>
                  </td>
                ))}
              </tr>
            ))}
          </tbody>
        </table>
      </div>

      {/* Legend */}
      <div className="flex items-center justify-center mt-4 gap-4">
        <div className="flex items-center gap-2">
          <div className="w-4 h-4 rounded bg-red-300"></div>
          <span className="text-xs">Loss</span>
        </div>
        <div className="flex items-center gap-2">
          <div className="w-4 h-4 rounded bg-gray-100"></div>
          <span className="text-xs">Breakeven</span>
        </div>
        <div className="flex items-center gap-2">
          <div className="w-4 h-4 rounded bg-green-300"></div>
          <span className="text-xs">Profit</span>
        </div>
      </div>
    </Card>
  );
}

interface Insight {
  id: string;
  type: 'behavioral' | 'risk' | 'pattern' | 'action';
  title: string;
  description: string;
  confidence: number; // 0-1
  actionability: 'quick_fix' | 'requires_attention' | 'strategic';
  suggestedAction: string;
}

interface InsightsPanelProps {
  insights: Insight[];
  onActionClick?: (insightId: string) => void;
}

export function InsightsPanel({ insights, onActionClick }: InsightsPanelProps) {
  // Sort by actionability: quick_fix first, then requires_attention, then strategic
  const sortedInsights = [...insights].sort((a, b) => {
    const priority = { quick_fix: 0, requires_attention: 1, strategic: 2 };
    return priority[a.actionability] - priority[b.actionability];
  }).slice(0, 3); // Top 3

  const getTypeIcon = (type: Insight['type']) => {
    switch (type) {
      case 'behavioral': return '🧠';
      case 'risk': return '⚠️';
      case 'pattern': return '📊';
      case 'action': return '🎯';
    }
  };

  const getActionabilityBadge = (actionability: Insight['actionability']) => {
    switch (actionability) {
      case 'quick_fix': 
        return <Badge variant="success">Quick Fix</Badge>;
      case 'requires_attention': 
        return <Badge variant="warning">Needs Attention</Badge>;
      case 'strategic': 
        return <Badge variant="outline">Strategic</Badge>;
    }
  };

  return (
    <Card>
      <div className="flex items-center justify-between mb-4">
        <h3 className="text-lg font-semibold">Actionable Insights</h3>
        <Link href="/analysis" className="text-sm text-primary hover:underline">
          View all →
        </Link>
      </div>

      <div className="space-y-4">
        {sortedInsights.map(insight => (
          <div key={insight.id} className="border rounded-lg p-4">
            <div className="flex items-start justify-between mb-2">
              <div className="flex items-center gap-2">
                <span className="text-lg">{getTypeIcon(insight.type)}</span>
                <span className="font-medium">{insight.title}</span>
              </div>
              {getActionabilityBadge(insight.actionability)}
            </div>
            
            <p className="text-sm text-muted-foreground mb-3">
              {insight.description}
            </p>
            
            <div className="flex items-center justify-between">
              <div className="text-xs text-muted-foreground">
                Confidence: {(insight.confidence * 100).toFixed(0)}%
              </div>
              <Button 
                size="sm" 
                onClick={() => onActionClick?.(insight.id)}
              >
                {insight.suggestedAction}
              </Button>
            </div>
          </div>
        ))}

        {sortedInsights.length === 0 && (
          <div className="text-center text-muted-foreground py-8">
            No insights available yet. Start logging trades to receive personalized recommendations.
          </div>
        )}
      </div>
    </Card>
  );
}
```

import { AIAertsWidget } from '@/components/dashboard/AIAertsWidget';
import { TradeStatsWidget } from '@/components/dashboard/TradeStatsWidget';
import { SessionHeatmap } from '@/components/dashboard/SessionHeatmap';
import { InsightsPanel } from '@/components/dashboard/InsightsPanel';
import { TradeQualityCard } from '@/components/analysis/TradeQualityCard';
import { BehavioralPatterns } from '@/components/analysis/BehavioralPatterns';
import { RiskMetricsCard } from '@/components/analysis/RiskMetricsCard';

export default async function AnalysisPage() {
  // Fetch dashboard data
  const response = await fetch(`${process.env.NEXT_PUBLIC_APP_URL}/api/analytics/dashboard?days=30`, {
    headers: { cookie: headers().get('cookie') || '' }
  });
  const dashboardData = await response.json();

  // Fetch detailed analysis
  const analysisResponse = await fetch(`${process.env.NEXT_PUBLIC_APP_URL}/api/ai/analyze`, {
    method: 'POST',
    headers: { 
      'Content-Type': 'application/json',
      cookie: headers().get('cookie') || ''
    },
    body: JSON.stringify({ 
      startDate: new Date(Date.now() - 30 * 24 * 60 * 60 * 1000).toISOString(),
      endDate: new Date().toISOString()
    })
  });
  const analysisData = await analysisResponse.json();

  return (
    <div className="space-y-6">
      {/* Header */}
      <div className="flex items-center justify-between">
        <div>
          <h1 className="text-2xl font-bold">AI Analysis</h1>
          <p className="text-muted-foreground">Trade quality, patterns, and insights</p>
        </div>
        <div className="flex gap-2">
          <DateRangePicker />
          <Button variant="outline">Export PDF</Button>
        </div>
      </div>

      {/* Dashboard Widgets */}
      <div className="grid grid-cols-1 md:grid-cols-2 gap-6">
        <AIAertsWidget alerts={dashboardData.aiAlerts} />
        <TradeStatsWidget 
          pairStats={dashboardData.tradeStats?.pairs || []} 
          timeStats={dashboardData.tradeStats?.times || []} 
        />
      </div>

      {/* Heatmap */}
      <SessionHeatmap data={dashboardData.heatmap || []} />

      {/* Insights */}
      <InsightsPanel insights={dashboardData.insights || []} />

      {/* Detailed Analysis Cards */}
      <div className="grid grid-cols-1 md:grid-cols-2 gap-6">
        <TradeQualityCard 
          trades={analysisData.analysis?.tradeAnalyses || []} 
        />
        <BehavioralPatterns 
          patterns={analysisData.analysis?.behavioralPatterns} 
        />
      </div>

      <RiskMetricsCard 
        metrics={analysisData.analysis?.aggregatedRisk} 
      />
    </div>
  );
}
