    const rankInfo = {

      Elite: { color: 'bg-yellow-500', next: null, points: 95 },

    interface EdgeScoreCardProps {
      score?: number
      rank?: keyof typeof rankInfo
      trend?: 'up' | 'down' | 'flat'

    export function EdgeScoreCard({ 
      score = 78, 
      rank = 'Consistent',
      trend = 'up' 
    }: EdgeScoreCardProps) {
      const currentRank = rankInfo[rank]
      const nextRank = currentRank.next
      const pointsToNext = nextRank ? rankInfo[nextRank as keyof typeof rankInfo].points - score : 0

        'Log at least 5 trades this week to improve your discipline score',
        'Review your journal entries for recurring patterns',
      ]

        // Check if user has completed tour
        const checkTourStatus = async () => {
          const { data: { user } } = await supabase.auth.getUser()
          
          if (user) {
            // Check user_metadata for tour completion
            const hasCompletedTour = user.user_metadata?.onboarding_completed
            if (!hasCompletedTour) {
              setRun(true)
            }
          } else {
            // Fallback to localStorage for non-authenticated state
            const localTour = localStorage.getItem('onboarding_completed')
            if (!localTour) {
              setRun(true)
            }
          }
        }

        checkTourStatus()
      }, [supabase])

      const handleJoyrideCallback = async (data: { status: string }) => {
        const { status } = data

        if ([STATUS.FINISHED, STATUS.SKIPPED].includes(status)) {
          setRun(false)
          
          // Mark tour as completed
          const { data: { user } } = await supabase.auth.getUser()
          
          if (user) {
            // Update user metadata in Supabase
            await supabase.auth.updateUser({
              data: { onboarding_completed: true }
            })
          } else {
            // Fallback to localStorage
            localStorage.setItem('onboarding_completed', 'true')
          }
        }
      }

      // Don't render on server
      if (typeof window === 'undefined') {
        return null
      }

      return (
        <Joyride
          run={run}
          steps={steps}
          continuous
          showProgress
          showSkipButton
          styles={{
            primaryColor: '#E53935',
            buttonNext: {
              backgroundColor: '#E53935',
              borderRadius: '6px',
            },
            buttonSkip: {
              color: '#1F2933',
            },
          }}
          locale={{
            last: 'Finish',
            skip: 'Skip',
          }}
          callback={handleJoyrideCallback}
          stepIndex={stepIndex}
          onStepChange={(step) => setStepIndex(step.index)}
        />
      )
